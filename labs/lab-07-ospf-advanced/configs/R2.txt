! ==========================================================
!  R2  - ABR (area 0 + area 26 stub)  [AS-BUILT]
!  e0/0 = backbone 10.0.0.0/24, area 0 (via switch)
!  e0/1 = to R6, 10.2.6.0/24, area 26 (stub)
!  Router ID: 2.2.2.2  |  OSPF priority 100 (DR)
! ==========================================================
hostname R2
!
interface Loopback1
 ip address 2.2.2.2 255.255.255.255
!
interface Ethernet0/0
 ip address 10.0.0.2 255.255.255.0
 ip ospf 10 area 0
 ip ospf priority 100
 no shutdown
!
interface Ethernet0/1
 ip address 10.2.6.2 255.255.255.0
 ip ospf 10 area 26
 ip ospf network point-to-point
 no shutdown
!
router ospf 10
 router-id 2.2.2.2
 area 26 stub
!
end
