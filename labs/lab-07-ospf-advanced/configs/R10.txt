hostname R10

interface Loopback1
 ip address 10.10.10.10 255.255.255.255
!
interface Loopback100

!
interface Loopback101

 ip ospf 10 area 48
!
interface Loopback102

!

 ip ospf network point-to-point

!
router ospf 10
 router-id 10.10.10.10
 area 108 virtual-link 8.8.8.8
!
end
