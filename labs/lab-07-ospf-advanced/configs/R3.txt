!  e0/0 = backbone 10.0.0.0/24, area 0 (via switch)
!  e0/1 = to R5, 10.3.5.0/24, area 35 (NSSA)
!  Router ID: 3.3.3.3  |  OSPF priority 80 (BDR)
! ==========================================================
hostname R3
!
interface Loopback1
 ip address 3.3.3.3 255.255.255.255
!
interface Ethernet0/0
 ip address 10.0.0.3 255.255.255.0
 ip ospf 10 area 0
 ip ospf priority 80
 no shutdown
!
interface Ethernet0/1
 ip address 10.3.5.3 255.255.255.0
 ip ospf 10 area 35
 ip ospf network point-to-point
 no shutdown
!
router ospf 10
 router-id 3.3.3.3
 area 35 nssa
!
end
