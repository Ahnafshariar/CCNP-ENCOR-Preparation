! ==========================================================
!  R4  - EIGRP edge, VPC8 gateway  [AS-BUILT]
!  e0/0 = to R2 (10.2.4.0/24, BW2000)
!  e0/1 = to R3 (10.3.4.0/24, BW1000)
!  e0/2 = to VPC8 LAN (10.4.0.0/24)
! ==========================================================
hostname R4
!
interface Ethernet0/0
 ip address 10.2.4.2 255.255.255.0
 bandwidth 2000
 no shutdown
!
interface Ethernet0/1
 ip address 10.3.4.2 255.255.255.0
 bandwidth 1000
 no shutdown
!
interface Ethernet0/2
 ip address 10.4.0.1 255.255.255.0
 no shutdown
!
router eigrp 2000
 network 10.2.4.2 0.0.0.0
 network 10.3.4.2 0.0.0.0
 network 10.4.0.1 0.0.0.0
!
end
