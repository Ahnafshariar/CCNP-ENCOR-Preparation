! ==========================================================
!  R1  - OSPF backbone router (area 0 only)  [AS-BUILT]
!  e0/0 = to R2 (10.1.2.0/24, area 0)
!  e0/1 = to R3 (10.1.3.0/24, area 0)
!  Router ID: 1.1.1.1
! ==========================================================
hostname R1
!
interface Ethernet0/0
 ip address 10.1.2.1 255.255.255.0
 no shutdown
!
interface Ethernet0/1
 ip address 10.1.3.1 255.255.255.0
 no shutdown
!
router ospf 10
 router-id 1.1.1.1
 network 10.1.2.1 0.0.0.0 area 0
 network 10.1.3.1 0.0.0.0 area 0
!
end
