! ==========================================================
!  R2  - VRF Lite (three isolated routing tables)  [AS-BUILT]
!  e0/0    = WAN trunk to R1 (subinterfaces per VRF)
!  e0/1    = LAN, VRF A (VPC8, 20.1.0.0/24)
!  e0/2    = LAN, VRF B (VPC6, 20.1.0.0/24)
!  e0/3    = LAN, VRF C (VPC7, 20.1.0.0/24)
! ==========================================================
hostname R2
!
ip vrf A
ip vrf B
ip vrf C
!
interface Ethernet0/0
 no ip address
 no shutdown
!
interface Ethernet0/0.1
 ip vrf forwarding A
 encapsulation dot1Q 10
 ip address 172.16.0.1 255.255.255.252
!
interface Ethernet0/0.2
 ip vrf forwarding B
 encapsulation dot1Q 20
 ip address 172.16.0.1 255.255.255.252
!
interface Ethernet0/0.3
 ip vrf forwarding C
 encapsulation dot1Q 30
 ip address 172.16.0.1 255.255.255.252
!
interface Ethernet0/1
 ip vrf forwarding A
 ip address 20.1.0.1 255.255.255.0
 no shutdown
!
interface Ethernet0/2
 ip vrf forwarding B
 ip address 20.1.0.1 255.255.255.0
 no shutdown
!
interface Ethernet0/3
 ip vrf forwarding C
 ip address 20.1.0.1 255.255.255.0
 no shutdown
!
ip route vrf A 10.1.0.0 255.255.255.0 172.16.0.2
ip route vrf B 10.1.0.0 255.255.255.0 172.16.0.2
ip route vrf C 10.1.0.0 255.255.255.0 172.16.0.2
!
end
