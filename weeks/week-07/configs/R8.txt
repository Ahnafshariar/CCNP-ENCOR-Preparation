!  Lo100-102 = 192.168.200-202.x (advertised in area 48)

hostname R8
!
interface Loopback1
 ip address 8.8.8.8 255.255.255.255
!
interface Loopback100
 ip address 192.168.200.8 255.255.255.255
 ip ospf 10 area 48
!
interface Loopback101
 ip address 192.168.201.8 255.255.255.255
 ip ospf 10 area 48
!
interface Loopback102

!
interface Ethernet0/0

!
interface Ethernet0/1
 ip address 10.8.10.8 255.255.255.0
 ip ospf 10 area 108
 ip ospf network point-to-point
 no shutdown
!
router ospf 10
 router-id 8.8.8.8
 area 48 virtual-link 4.4.4.4
 area 108 virtual-link 10.10.10.10
!
